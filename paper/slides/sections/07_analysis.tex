\section{分析与案例}

\begin{frame}{两个关键失败原因}
  \begin{columns}[T,onlytextwidth]
    \column{0.48\textwidth}
    \begin{block}{原因一：异构格式噪声}
      \small
      \begin{tabular}{@{}lcc@{}}
        \toprule
        形态 & 平均长度 & 特殊符号比例 \\ \midrule
        纯文本 & 600.97 & 0.47\% \\
        异构序列化 & 53.07 & \hl{8.78\%} \\
        \bottomrule
      \end{tabular}
      \vspace{0.4em}

      \begin{itemize}
        \item 信息更碎片化
        \item 符号密度更高
        \item 不利于 embedding 与生成对齐
      \end{itemize}
    \end{block}

    \column{0.52\textwidth}
    \begin{alertblock}{原因二：启发式 CoVe 缺乏 sweet spot}
      \small
      \begin{tabular}{@{}lccc@{}}
        \toprule
        配置 & NAR & FRR & Unsafe \\ \midrule
        Soft   & 0.5\%  & 0.5\%  & \hl{44.0\%} \\
        Strict & \hl{98.5\%} & \hl{98.5\%} & 0.5\% \\
        \bottomrule
      \end{tabular}
      \vspace{0.4em}

      \begin{itemize}
        \item 软阈值：漏放严重
        \item 严阈值：误杀严重
      \end{itemize}
    \end{alertblock}
  \end{columns}
\end{frame}

\begin{frame}{案例分析：方法何时有效，何时失效}
  \begin{columns}[T,onlytextwidth]
    \column{0.5\textwidth}
    \begin{exampleblock}{成功案例：SpaceX 多跳问题}
      \begin{itemize}
        \item 首轮命中 Falcon 9 与 SpaceX
        \item PRF 扩展为 \texttt{SpaceX founder year}
        \item 二轮补齐“Elon Musk / 2002”
      \end{itemize}
      \hl{说明：Adaptive 对 bridge 类问题有效。}
    \end{exampleblock}

    \column{0.5\textwidth}
    \begin{alertblock}{失败案例：Ermengarde of Anjou}
      \begin{itemize}
        \item 生成答案为 \textbf{Anjou}，本身正确
        \item 图谱中证据写作 \texttt{Angers (in Anjou)}
        \item CoVe 仅做词面比对，误判为冲突
      \end{itemize}
      \hl{说明：当前验证器缺乏语义层面的实体对齐能力。}
    \end{alertblock}
  \end{columns}
\end{frame}
