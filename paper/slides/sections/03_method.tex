\section{方法与系统架构}

\begin{frame}{系统总体架构}
  \begin{columns}[T,onlytextwidth]
    \column{0.64\textwidth}
    \centering
    \includegraphics[width=\linewidth]{rag_architecture_flow.png}

    \column{0.36\textwidth}
    \begin{block}{整体流程}
      \begin{itemize}
        \item 检索与重排
        \item 低置信触发补检索
        \item 证据组织后生成答案
        \item 生成后再做验证与拒答
      \end{itemize}
    \end{block}
    \begin{exampleblock}{两个反馈环}
      \begin{itemize}
        \item \hl{低置信} $\rightarrow$ 扩展查询
        \item \hl{证据冲突} $\rightarrow$ No-Answer
      \end{itemize}
    \end{exampleblock}
  \end{columns}
\end{frame}

\begin{frame}{方法设计：三项关键机制}
  \begin{columns}[T,onlytextwidth]
    \column{0.33\textwidth}
    \begin{block}{\numbox{1}\ EvidenceUnit}
      以 \texttt{content / source / metadata} 统一三类证据表示，支持\hl{同一检索与重排接口}。
    \end{block}

    \column{0.33\textwidth}
    \begin{block}{\numbox{2}\ Adaptive PRF}
      若 Top-1 重排分低于阈值 $\tau$，则从高分证据中抽取关键词，执行\hl{二次检索 + RRF 融合}。
    \end{block}

    \column{0.33\textwidth}
    \begin{block}{\numbox{3}\ CoVe 验证}
      将答案拆成原子声明，与证据做支持度比对；若证据不足，则返回\hl{拒答}而非冒险作答。
    \end{block}
  \end{columns}

  \vspace{0.5em}
  \begin{exampleblock}{实现策略}
    使用轻量 embedding + CrossEncoder 重排 + 启发式生成后端，尽量把实验差异归因到检索与验证模块本身。
  \end{exampleblock}
\end{frame}

\begin{frame}[fragile]{异构接入与补检索机制}
  \begin{columns}[T,onlytextwidth]
    \column{0.48\textwidth}
    \begin{block}{异构序列化示例}
      \textbf{表格行：}
      \begin{lstlisting}[style=shellstyle, basicstyle=\ttfamily\tiny]
Table: Tech Founders
Company: SpaceX, Founder: Elon Musk
Year: 2002
      \end{lstlisting}
      \textbf{图谱三元组：}
      \begin{lstlisting}[style=shellstyle, basicstyle=\ttfamily\tiny]
Graph Fact: Elon Musk
 -> founder_of -> SpaceX
      \end{lstlisting}
    \end{block}

    \column{0.52\textwidth}
    \begin{block}{Adaptive PRF 判定逻辑}
      \[
        s_1 < \tau \Rightarrow \text{Expand query and retrieve again}
      \]
      其中 $s_1$ 为首轮 Top-1 重排分，RRF 融合公式为：
      \[
        \mathrm{RRF}(d)=\sum \frac{1}{60+\mathrm{rank}(d)}
      \]
    \end{block}
    \begin{alertblock}{设计动机}
      不直接“多检索一些”，而是只在\hl{证据不足时}付出额外成本。
    \end{alertblock}
  \end{columns}
\end{frame}
