\begin{frame}{主消融结果：Baseline 反而最稳定}
  \begin{columns}[T,onlytextwidth]
    \column{0.56\textwidth}
    \centering
    \includegraphics[width=\linewidth]{ablation_f1.png}

    \column{0.44\textwidth}
    \begin{block}{关键数值（HotpotQA, N=7405）}
      \small
      \begin{tabular}{@{}lcc@{}}
        \toprule
        配置 & F1 & NAR \\ \midrule
        A   & \hl{6.16} & 0.0\% \\
        A2  & 6.16 & 0.0\% \\
        A3  & 5.76 & 13.5\% \\
        B   & 4.59 & 0.0\% \\

    \begin{alertblock}{结论}
      \begin{itemize}
        \item \hl{异构接入未带来稳定 F1 增益}
        \item strict CoVe 导致\hl{大规模拒答}
      \end{itemize}
    \end{alertblock}
  \end{columns}
\end{frame}

\begin{frame}{成本与安全：模块收益并非免费}
  \begin{columns}[T,onlytextwidth]
    \column{0.5\textwidth}
    \centering
    \includegraphics[width=\linewidth]{ablation_cost_latency.png}\\[0.3em]
    \footnotesize\color{SoftGray} Token 与延迟

    \column{0.5\textwidth}
    \centering
    \includegraphics[width=\linewidth]{ablation_safety.png}\\[0.3em]
    \footnotesize\color{SoftGray} 拒答率（NAR）
  \end{columns}

  \vspace{0.5em}
  \begin{exampleblock}{解读}
    \begin{itemize}
      \item A2 相比 A：延迟上升，但 F1 基本不变。
      \item B/C 序列化更短，Token 降低，但 \hl{格式噪声}导致效果下降。
      \item D 的安全性建立在\hl{几乎全拒答}之上，不具备实际可用性。
    \end{itemize}
  \end{exampleblock}
\end{frame}

\begin{frame}{Adaptive 的价值：体现在证据覆盖率}
  \begin{columns}[T,onlytextwidth]
    \column{0.52\textwidth}
    \centering
    \includegraphics[width=\linewidth]{cross_dataset_support_hit.png}

    \column{0.48\textwidth}
    \begin{block}{paired bootstrap 95\% CI}
      \small
      \begin{tabular}{@{}lc@{}}
        \toprule
        数据集 & $\Delta$ SupportAllHit@5 \\ \midrule
        HotpotQA & \hl{+5.0 pp} \\
        2Wiki    & \hl{+7.5 pp} \\
        \bottomrule
      \end{tabular}
    \end{block}
    \begin{exampleblock}{分桶观察}
      bridge 类问题增益最明显：\hl{+7.92 pp}；comparison 类增益接近 0。
    \end{exampleblock}
    \begin{alertblock}{含义}
      Adaptive 更像\hl{补齐第二跳证据}的机制，而不是直接提升最终答案 F1 的机制。
    \end{alertblock}
  \end{columns}
\end{frame}
