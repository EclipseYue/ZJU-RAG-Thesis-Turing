% ---- 6.1 主消融数值表 -------------------------------------------
\begin{frame}{主消融：HotpotQA dev 全量 (N=7405)}
  \footnotesize
  \centering
  \begin{tabular}{@{}lccccc@{}}
    \toprule
    \rowcolor{ZjuBlue!10}
    \textbf{ID} & \textbf{配置} & \textbf{F1} & \textbf{NAR} & \textbf{Tokens} & \textbf{Latency (ms)} \\ \midrule
    \rowcolor{AccentTeal!8}
    A   & Baseline                    & \hlt{6.16} & 0.0\% & 1012.47 & 236.05 \\
    A2  & + Adaptive only             & 6.16  & 0.0\%   & 1041.02 & 301.47 \\
    A3  & + CoVe only                 & 5.76  & 13.52\% & 552.26  & 170.54 \\
    B   & + Hetero only               & 4.59  & 0.0\%   & 716.85  & 281.04 \\
    C   & + Hetero + Adaptive         & 4.59  & 0.0\%   & 743.60  & 291.51 \\
    \rowcolor{ZjuRed!10}
    D   & Full (strict CoVe)          & \hl{0.06} & \hl{97.2\%} & 743.60  & 291.31 \\
    \bottomrule
  \end{tabular}

  \vspace{0.6em}
  \begin{alertblock}{\faSearch\ 三个反直觉发现}
    \footnotesize
    \begin{enumerate}
      \item \hl{Baseline 反而最稳定}——在启发式生成端控制下，加模块普遍\hl{未带来 F1 增益}。
      \item \hl{Hetero 拉低 F1 (6.16 → 4.59)}——\hlb{格式噪声}超过了"多源信息"带来的潜在收益。
      \item \hl{strict CoVe 几乎拒答全部 (97.2\%)}——把 F1 拉到接近 0；\hlb{soft / strict 之间无 sweet spot}。
    \end{enumerate}
  \end{alertblock}
\end{frame}

    \hlb{comparison 类问题}本身两个对比对象都已在 Top-$k$，补检索几乎无新增信息。
  \end{exampleblock}

      \item \hlb{阈值不敏感}：SupportAllHit@5 在 $\tau \in [0.7, 0.95]$ 上恒为 66.0；\\
            但延迟从 \hl{1015.64ms 降至 919.62ms} —— \hlt{更保守的阈值反而更省}。
      \item Pareto 上 \hlb{Adaptive Only} 与 \hlb{Baseline} 距离很近 ——
            \hl{当前形态下，"加模块"的边际收益有限}。
    \end{itemize}
