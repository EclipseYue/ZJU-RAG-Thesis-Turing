\section{研究背景与动机}

% ---- 1.1 大背景 -------------------------------------------------
\begin{frame}{LLM 时代的"知识鸿沟"}
  \begin{columns}[T,onlytextwidth]
    \column{0.55\textwidth}
    \begin{block}{大语言模型的能力边界}
      \begin{itemize}
        \item \hlb{参数化知识}：训练截止后即静态，难以更新。
        \item \hlb{长尾事实薄弱}：稀有实体、最新事件易出错。
        \item \hlb{幻觉风险}：生成"看似合理但事实错误"的内容。
        \item \hlb{推理可控性差}：缺乏可追溯的中间证据。
      \end{itemize}
    \end{block}

    \column{0.45\textwidth}
    \begin{alertblock}{应对范式：检索增强生成 (RAG)}
      在生成前\hl{显式检索外部知识}，将证据注入上下文：
      \begin{itemize}
        \item 缓解幻觉
        \item 支持知识更新
        \item 提供可追溯证据链
      \end{itemize}
    \end{alertblock}
  \end{columns}

  \vspace{0.5em}
  \footnotesize\color{SoftGray}
  代表工作：Lewis et al.\ (2020) RAG；Guu et al.\ (2020) REALM；
  Asai et al.\ (2023) Self-RAG；Trivedi et al.\ (2023) IRCoT。
\end{frame}

% ---- 1.2 RAG 的真实挑战 ------------------------------------------
\begin{frame}{然而，真实场景下 RAG 仍面临三重挑战}
  \begin{columns}[T,onlytextwidth]
    \column{0.33\textwidth}
    \begin{block}{\numbox{1}\ 异构融合困难}
      知识源不只是文本：还有\hl{表格}、\hl{知识图谱}、半结构化数据。\\[0.3em]
      不同模态如何\hlb{统一表示}并进入同一检索流程？
    \end{block}

    \column{0.33\textwidth}
    \begin{block}{\numbox{2}\ 复杂跨文档推理}
      多跳问题、桥接实体、否定与约束\ldots\\[0.3em]
      单轮检索易\hl{遗漏第二跳证据}；需要\hlb{按需补检索}而非简单地"检索更多"。
    \end{block}

    \column{0.33\textwidth}
    \begin{block}{\numbox{3}\ 推理缺乏可信验证}
      "看起来对"不等于"被证据支持"。\\[0.3em]
      何时该\hl{拒答}（No-Answer）？\hlb{如何在拒答与可用性之间取得平衡}？
    \end{block}
  \end{columns}

  \vspace{0.8em}
  \begin{exampleblock}{本研究关注的命题}
    把 RAG 视为\hl{兼顾正确率、成本与安全性的序列决策过程}，
    在统一框架内显式权衡\hlt{效果 / Token 与延迟成本 / 拒答安全性}三者矛盾。
  \end{exampleblock}
\end{frame}

% ---- 1.4 相关工作四条线 ------------------------------------------
\begin{frame}{相关工作概览}
  \begin{columns}[T,onlytextwidth]
    \column{0.5\textwidth}
    \begin{block}{① 纯文本检索基础}
      \footnotesize
      \begin{itemize}
        \item BM25、DPR (Karpukhin 2020)、ColBERT
        \item CrossEncoder Rerank (BGE, MonoT5)
        \item RRF 融合 (Cormack 2009)
      \end{itemize}
    \end{block}
    \begin{block}{② 主动检索 / 迭代检索}
      \footnotesize
      \begin{itemize}
        \item FLARE (Jiang 2023)：低置信度触发再检索
        \item IRCoT (Trivedi 2023)：CoT 与检索交替
        \item Self-RAG (Asai 2023)：反射式检索
      \end{itemize}
    \end{block}

    \column{0.5\textwidth}
    \begin{block}{③ 结构化 / 异构融合}
      \footnotesize
      \begin{itemize}
        \item HybridQA、OTT-QA：表格与文本联合 QA
        \item ToG (Sun 2024)：图谱思维链
        \item HippoRAG (Gutiérrez 2024)：仿海马体索引
      \end{itemize}
    \end{block}
    \begin{block}{④ 后验自检 / 可信验证}
      \footnotesize
      \begin{itemize}
        \item CoVe (Dhuliawala 2023)：自我验证链
        \item FactScore (Min 2023)：原子事实评估
        \item TRACE (Fang 2024)：证据链可追溯
      \end{itemize}
    \end{block}
  \end{columns}

  \vspace{0.4em}
  \begin{exampleblock}{本研究的定位}
    \footnotesize
    \hlt{不是提出全新范式}，而是把"异构接入 + 主动补检索 + 后验验证"\hlb{置于同一可复现实验下}，
    定量观察其耦合表现与失败模式，揭示 RAG 中尚未被广泛披露的\hl{副作用与权衡}。
  \end{exampleblock}
\end{frame}
